// 适用于 3.0.0 版本的公共分享函数
export function createShareConfig(config = {}) {

        },
        fail: (err) => {

### 方案3：使用 mixin 实现复用

**步骤1：创建分享 mixin**

```
// mixins/share.js
export default {
  data() {
    return {
      shareConfig: {
        title: '可我不敌心软~',
        path: '/pages/index/index',
        imageUrl: 'https://cdn.aioveu.com/aioveu-server/avatar/avatar.png',
        query: 'key=value'
      }
    }
  },
  
  onShareAppMessage() {
    return {
      title: this.shareConfig.title,
      path: this.shareConfig.path + (this.shareConfig.query ? `?${this.shareConfig.query}` : ''),
      imageUrl: this.shareConfig.imageUrl,
      success: (res) => {
        console.log("分享成功", res)
      },
      fail: (err) => {
        console.log("分享失败", err)
      }
    }
  },
  
  onShareTimeline() {
    return {
      title: this.shareConfig.title,
      query: this.shareConfig.query,
      imageUrl: this.shareConfig.imageUrl,
      success: (res) => {
        console.log("分享到朋友圈成功", res)
      },
      fail: (err) => {
        console.log("分享失败", err)
      }
    }
  },
  
  methods: {
    updateShareConfig(config) {
      this.shareConfig = { ...this.shareConfig, ...config }
    }
  }
}
```

**步骤2：在页面中使用 mixin**

```
<!-- pages/index/index.vue -->

      })
    }
  },
  
  data() {
    return {
      // 页面自己的数据
    }
  }
}
</script>
```

## 总结

**您的 uni-app 3.0.0 版本不支持 `definePageConfig`**，应该：
