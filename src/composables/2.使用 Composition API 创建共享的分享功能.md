在 Vue 3 的 uni-app 中，可以使用 Composition API 创建共享的分享功能。以下是几种实现方式：

## 方法1：使用 Composition API 创建共享 Hook（推荐）

**步骤1：创建分享 Hook**

```
// composables/useShare.js
import { ref, onMounted, onUnmounted } from 'vue'

export function useShare(defaultConfig = {}) {
  // 默认分享配置
  const defaultShareConfig = {
    title: '可我不敌心软~',
    path: '/pages/index/index',
    imageUrl: 'https://cdn.aioveu.com/aioveu-server/avatar/avatar.png',
    query: 'key=value'
  }
  
  // 合并配置
  const shareConfig = ref({
    ...defaultShareConfig,
    ...defaultConfig
  })
  
  // 分享成功回调
  const onShareSuccess = (res) => {
    console.log('分享成功', res)
  }
  
  // 分享失败回调
  const onShareFail = (err) => {
    console.log('分享失败', err)
  }
  
  // 朋友圈分享成功回调
  const onShareTimelineSuccess = (res) => {
    console.log('分享到朋友圈成功', res)
  }
  
  // 朋友圈分享失败回调
  const onShareTimelineFail = (err) => {
    console.log('分享到朋友圈失败', err)
  }
  
  // 设置分享的方法
  const setupShare = (customConfig = {}) => {
    // 更新配置
    if (customConfig) {
      shareConfig.value = { ...shareConfig.value, ...customConfig }
    }
    
    // 设置分享给朋友
    uni.onShareAppMessage(() => ({
      title: shareConfig.value.title,
      path: shareConfig.value.path + (shareConfig.value.query ? `?${shareConfig.value.query}` : ''),
      imageUrl: shareConfig.value.imageUrl,
      success: onShareSuccess,
      fail: onShareFail
    }))

    // 设置分享到朋友圈
    uni.onShareTimeline(() => ({
      title: shareConfig.value.title,
      query: shareConfig.value.query,
      imageUrl: shareConfig.value.imageUrl,
      success: onShareTimelineSuccess,
      fail: onShareTimelineFail
    }))
  }
  
  // 更新分享配置
  const updateShareConfig = (newConfig) => {
    shareConfig.value = { ...shareConfig.value, ...newConfig }
    setupShare()
  }
  
  return {
    shareConfig,
    setupShare,
    updateShareConfig,
    onShareSuccess,
    onShareFail,
    onShareTimelineSuccess,
    onShareTimelineFail
  }
}
```

**步骤2：在页面组件中使用**

```
<template>
  <view>
    <!-- 页面内容 -->
  </view>
</template>

<script setup>

    icon: 'success'
  })
}

```

## 方法2：创建可复用的分享组件

**步骤1：创建分享组件**

```
<!-- components/GlobalShare.vue -->
<template>
  <!-- 这是一个无UI组件，只提供功能 -->
</template>

<script setup>
import { onMounted, onUnmounted, defineProps, watch } from 'vue'

const props = defineProps({
  // 分享配置
  config: {
    type: Object,
    default: () => ({
      title: '可我不敌心软~',
      path: '/pages/index/index',
      imageUrl: 'https://cdn.aioveu.com/aioveu-server/avatar/avatar.png',
      query: 'key=value'
    })
  },
  // 是否启用分享
  enable: {
    type: Boolean,
    default: true
  }
})

// 事件定义
const emit = defineEmits([
  'share-success',
  'share-fail',
  'timeline-success',
  'timeline-fail'
])

// 设置分享
const setupShare = () => {
  if (!props.enable) return
  
  // 分享给朋友
  uni.onShareAppMessage(() => ({
    title: props.config.title,
    path: props.config.path + (props.config.query ? `?${props.config.query}` : ''),
    imageUrl: props.config.imageUrl,
    success: (res) => {
      console.log("分享成功", res)
      emit('share-success', res)
    },
    fail: (err) => {
      console.log("分享失败", err)
      emit('share-fail', err)
    }
  }))

  // 分享到朋友圈
  uni.onShareTimeline(() => ({
    title: props.config.title,
    query: props.config.query,
    imageUrl: props.config.imageUrl,
    success: (res) => {
      console.log("分享到朋友圈成功", res)
      emit('timeline-success', res)
    },
    fail: (err) => {
      console.log("分享到朋友圈失败", err)
      emit('timeline-fail', err)
    }
  }))
}

// 监听配置变化
watch(() => props.config, () => {
  setupShare()
}, { deep: true })

onMounted(() => {
  setupShare()
})

onUnmounted(() => {
  // 如果需要清理，可以在这里处理
})
</script>
```

## 方法3：创建全局分享插件

**步骤1：创建分享插件**

```
// plugins/share.js
export default {
  install(app, options = {}) {
    // 默认配置
    const defaultConfig = {
      title: '可我不敌心软~',
      path: '/pages/index/index',
      imageUrl: 'https://cdn.aioveu.com/aioveu-server/avatar/avatar.png',
      query: 'key=value'
    }
    
    // 全局分享方法
    app.config.globalProperties.$setupShare = function(customConfig = {}) {
      const config = { ...defaultConfig, ...customConfig }
      
      // 分享给朋友
      uni.onShareAppMessage(() => ({
        title: config.title,
        path: config.path + (config.query ? `?${config.query}` : ''),
        imageUrl: config.imageUrl,
        success: (res) => {
          console.log("分享成功", res)
          this.$emit && this.$emit('share-success', res)
        },
        fail: (err) => {
          console.log("分享失败", err)
          this.$emit && this.$emit('share-fail', err)
        }
      }))

      // 分享到朋友圈
      uni.onShareTimeline(() => ({
        title: config.title,
        query: config.query,
        imageUrl: config.imageUrl,
        success: (res) => {
          console.log("分享到朋友圈成功", res)
          this.$emit && this.$emit('share-timeline-success', res)
        },
        fail: (err) => {
          console.log("分享到朋友圈失败", err)
          this.$emit && this.$emit('share-timeline-fail', err)
        }
      }))
    }
    
    // 全局分享配置
    app.config.globalProperties.$shareConfig = defaultConfig
  }
}
```

**步骤2：注册插件**

```
// main.js
import { createApp } from 'vue'
import App from './App.vue'
import sharePlugin from './plugins/share.js'

const app = createApp(App)

// 安装分享插件
app.use(sharePlugin)

app.mount('#app')
```

## 方法4：使用自定义指令（最简洁）

  mounted(el, binding) {

      title: '可我不敌心软~',
      path: '/pages/index/index',
      imageUrl: 'https://cdn.aioveu.com/aioveu-server/avatar/avatar.png',

    const shareConfig = { ...defaultConfig, ...config }

    uni.onShareAppMessage(() => ({
      title: shareConfig.title,
      path: shareConfig.path + (shareConfig.query ? `?${shareConfig.query}` : ''),
      imageUrl: shareConfig.imageUrl,
      success: (res) => console.log("分享成功", res),
      fail: (err) => console.log("分享失败", err)
    }))

    uni.onShareTimeline(() => ({
      title: shareConfig.title,
      query: shareConfig.query,
      imageUrl: shareConfig.imageUrl,
      success: (res) => console.log("分享到朋友圈成功", res),
      fail: (err) => console.log("分享到朋友圈失败", err)
    }))
  }
}

## 推荐使用

**推荐使用方法1（Composition API Hook）**，原因：

1. 符合 Vue 3 的设计理念
2. 类型友好，易于维护
3. 逻辑清晰，可复用性强
4. 便于测试和调试

使用 Hook 方式可以让分享逻辑与组件完全解耦，便于在多个页面中复用，同时也便于维护和扩展。
