// utils/page-config.js
// 模拟 definePageConfig 功能
export function definePageConfigCompatible(config) {
  // 获取当前页面
  const pages = getCurrentPages()
  const currentPage = pages[pages.length - 1]
  
  if (currentPage) {
    // 将配置应用到页面
    Object.assign(currentPage, config)
  }
  
  return config
}

### 方案2：创建页面配置装饰器

```
// decorators/page.js
export function Page(config) {
  return function(target) {
    // 将配置合并到组件选项
    const originalExport = target.prototype || target
    
    Object.keys(config).forEach(key => {
      if (typeof originalExport[key] === 'function' && typeof config[key] === 'function') {
        // 合并函数
        const originalFn = originalExport[key]
        originalExport[key] = function(...args) {
          return config[key].call(this, ...args) || originalFn.call(this, ...args)
        }
      } else {
        // 直接覆盖
        originalExport[key] = config[key]
      }
    })
    
    return target
  }
}
```

使用方式：

```
<template>
  <view>
    <!-- 页面内容 -->
  </view>
</template>

<script>
import { Page } from '@/decorators/page'

@Page({
  onShareAppMessage() {
    return {
      title: '可我不敌心软~',
      path: '/pages/index/index',
      imageUrl: 'https://cdn.aioveu.com/aioveu-server/avatar/avatar.png'
    }
  }
})
export default {
  data() {
    return {
      // 页面数据
    }
  }
}
</script>
```
