根据您的代码，我发现了几个问题。当令牌失效时，您的跳转代码存在问题：

## 🔍 问题分析

1. **跳转路径错误**：`"/packageA/pages/login/index"`这个路径可能在 uni-app 中不存在或不正确
2. **跳转时机问题**：在 `reLaunch`后没有结束 Promise，可能导致后续代码继续执行
3. **缺少错误处理**：跳转失败时没有备选方案

## 🚀 修复方案

```
// 令牌过期处理函数
let isTokenExpiredHandling = false; // 防止重复处理

function handleTokenExpired() {
  if (isTokenExpiredHandling) {
    console.log("⚠️ 令牌过期处理正在进行中，跳过重复处理");
    return;
  }
  
  isTokenExpiredHandling = true;
  console.log("🔄 开始处理令牌过期");
  
  // 1. 清除所有缓存
  try {
    clearAll();
  } catch (e) {
    console.warn("清除缓存失败:", e);
  }
  
  // 2. 显示提示
  setTimeout(() => {
    uni.showToast({
      title: "登录已过期，请重新登录",
      icon: "none",
      duration: 2000
    });
  }, 100);
  
  // 3. 延迟跳转，避免冲突
  setTimeout(() => {
    safeNavigateToLogin(() => {
      isTokenExpiredHandling = false; // 处理完成
    });
  }, 1500);
}

// 安全跳转到登录页
function safeNavigateToLogin(onComplete?: () => void) {
  // 获取当前页面栈
  const pages = getCurrentPages();
  const currentPage = pages.length > 0 ? pages[pages.length - 1] : null;
  const currentRoute = currentPage ? currentPage.route : '';
  
  console.log("📍 当前页面路由:", currentRoute);
  console.log("📍 页面栈深度:", pages.length);
  
  // 检查是否已经在登录页
  if (currentRoute && (
    currentRoute.includes('login') || 
    currentRoute.includes('Login') ||
    currentRoute === 'packageA/pages/login/index'
  )) {
    console.log("✅ 已经在登录页，不重复跳转");
    if (onComplete) onComplete();
    return;
  }
  
  // 定义可能的登录页路径
  const loginPaths = [
    "/pages/login/index",  // 常见路径1
    "/pages/user/login",   // 常见路径2
    "/packageA/pages/login/index", // 您原来的路径
    "/pages/auth/login",   // 常见路径3
    "/pages/mine/login"    // 常见路径4
  ];
  
  // 尝试跳转
  const tryNavigate = (index = 0) => {
    if (index >= loginPaths.length) {
      console.error("❌ 所有登录页路径都尝试失败");
      // 显示提示让用户手动操作
      uni.showModal({
        title: "提示",
        content: "登录状态已过期，请手动返回登录页面",
        showCancel: false,
        confirmText: "知道了",
        success: () => {
          if (onComplete) onComplete();
        }
      });
      return;
    }
    
    const loginUrl = loginPaths[index];
    console.log(`🔄 尝试跳转路径 ${index + 1}:`, loginUrl);
    
    // 设置跳转超时
    const navigateTimeout = setTimeout(() => {
      console.warn(`⏰ 跳转超时，尝试下一个路径: ${loginUrl}`);
      tryNavigate(index + 1);
    }, 3000);
    
    // 先尝试 reLaunch
    uni.reLaunch({
      url: loginUrl,
      success: () => {
        console.log(`✅ 成功跳转到登录页: ${loginUrl}`);
        clearTimeout(navigateTimeout);
        if (onComplete) onComplete();
      },
      fail: (err) => {
        clearTimeout(navigateTimeout);
        console.warn(`⚠️ reLaunch 跳转失败: ${loginUrl}`, err);
        
        // 如果 reLaunch 失败，尝试 navigateTo
        uni.navigateTo({
          url: loginUrl,
          success: () => {
            console.log(`✅ navigateTo 成功: ${loginUrl}`);
            if (onComplete) onComplete();
          },
          fail: () => {
            console.warn(`⚠️ navigateTo 也失败: ${loginUrl}`);
            // 尝试下一个路径
            tryNavigate(index + 1);
          }
        });
      }
    });
  };
  
  // 开始尝试跳转
  tryNavigate();
}
```

### 方案3：修改原代码的令牌失效处理部分

```
// 令牌失效或过期处理
else if (resData.code === ResultCodeEnum.TOKINVALID) {

// 同步处理函数
function handleTokenExpiredSync() {
  try {
    // 1. 清除缓存
    clearAll();
    
    // 2. 检查当前页面
    const pages = getCurrentPages();
    if (pages.length === 0) {
      // 如果是应用启动时，可能需要不同的处理
      console.log("应用启动时令牌过期");
      return;
    }
    
    const currentPage = pages[pages.length - 1];
    const currentRoute = currentPage.route || '';
    
    // 3. 如果已经在登录相关页面，不跳转
    if (currentRoute.includes('login') || 
        currentRoute.includes('auth') || 
        currentRoute.includes('Login')) {
      console.log("已在登录相关页面，不跳转");
      return;
    }
    
    // 4. 显示提示
    uni.showToast({
      title: "登录已过期，请重新登录",
      icon: "none",
      duration: 2000
    });
    
    // 5. 延迟跳转
    setTimeout(() => {
      // 优先尝试您的原始路径
      const loginUrl = "/packageA/pages/login/index";
      
      uni.reLaunch({
        url: loginUrl,
        success: () => {
          console.log("✅ 成功跳转到登录页");
        },
        fail: (reLaunchErr) => {
          console.error("❌ reLaunch 跳转失败:", reLaunchErr);
          
          // 备选方案1：尝试其他常见路径
          const alternativePaths = [
            "/pages/login/index",
            "/pages/user/login",
            "/pages/auth/login"
          ];
          
          for (const path of alternativePaths) {
            try {
              uni.navigateTo({
                url: path,
                fail: () => {
                  // 继续尝试下一个
                }
              });
              break;
            } catch (e) {
              console.warn(`跳转失败: ${path}`, e);
            }
          }
          
          // 如果所有跳转都失败，显示手动操作提示
          uni.showModal({
            title: "提示",
            content: "请手动返回登录页面",
            showCancel: false,
            confirmText: "确定"
          });
        }
      });
    }, 2000);
    
  } catch (error) {
    console.error("处理令牌过期时出错:", error);
  }
}
```

```
// 在 request.js 中
import { eventBus } from '@/utils/eventBus';
```

## 🎯 关键点

1. **确认登录页路径**：首先确保 `"/packageA/pages/login/index"`这个路径是真实存在的

4. **避免重复处理**：添加标记防止多次跳转
