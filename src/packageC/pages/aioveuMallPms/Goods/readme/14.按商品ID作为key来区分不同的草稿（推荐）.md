你需要根据**商品ID**来区分不同的草稿。有三种方案：

## 🎯 方案一：按商品ID作为key（推荐）

```
// GoodsStock.vue
const DRAFT_KEY_PREFIX = 'goods_stock_draft_';

// 获取商品ID
const goodsId = computed(() => goodsInfo.value.id);
const draftKey = computed(() => `${DRAFT_KEY_PREFIX}${goodsId.value || 'new'}`);

// 保存草稿
const saveDraftToLocal = () => {
  if (!goodsId.value) {
    console.log('新增商品，不保存草稿');
    return;
  }
  
  uni.setStorageSync(draftKey.value, {
    goodsId: goodsId.value,
    specForm: specForm.value,
    skuForm: skuForm.value,
    timestamp: Date.now(),
  });
  console.log(`✅ 草稿已保存: ${goodsId.value}`);
};

// 加载时检查草稿
onMounted(() => {
  if (!goodsId.value) return;
  
  const draft = uni.getStorageSync(draftKey.value);
  if (draft && draft.goodsId === goodsId.value) {
    uni.showModal({
      title: "检测到未保存的草稿",
      content: "是否恢复之前的编辑内容？",
      success: (res) => {
        if (res.confirm) {
          specForm.value = draft.specForm;
          skuForm.value = draft.skuForm;
          console.log(`✅ 草稿已恢复: ${goodsId.value}`);
        }
        // 无论是否恢复，都清除草稿
        uni.removeStorageSync(draftKey.value);
      }
    });
  }
});
```

## 🎯 方案二：存储对象，key为商品ID

```

onMounted(() => {

          skuForm.value = draft.skuForm;

      }
    });
  }
});

## 🎯 方案三：带自动清理功能

```
// GoodsStock.vue
const DRAFT_KEY_PREFIX = 'goods_stock_draft_';
const MAX_DRAFT_AGE = 24 * 60 * 60 * 1000; // 24小时

// 保存草稿
const saveDraftToLocal = () => {
  if (!goodsId.value) return;
  
  const draftData = {
    goodsId: goodsId.value,
    specForm: specForm.value,
    skuForm: skuForm.value,
    timestamp: Date.now(),
  };
  
  uni.setStorageSync(`${DRAFT_KEY_PREFIX}${goodsId.value}`, draftData);
  console.log(`✅ 草稿已保存: ${goodsId.value}`);
  
  // 自动清理过期的草稿
  cleanupExpiredDrafts();
};

// 清理过期草稿
const cleanupExpiredDrafts = () => {
  const now = Date.now();
  const keys = uni.getStorageInfoSync().keys || [];
  
  keys.forEach(key => {
    if (key.startsWith(DRAFT_KEY_PREFIX)) {
      const draft = uni.getStorageSync(key);
      if (draft && now - draft.timestamp > MAX_DRAFT_AGE) {
        uni.removeStorageSync(key);
        console.log(`🗑️ 清理过期草稿: ${draft.goodsId}`);
      }
    }
  });
};

// 加载时检查草稿
onMounted(() => {
  if (!goodsId.value) return;
  
  const draft = uni.getStorageSync(`${DRAFT_KEY_PREFIX}${goodsId.value}`);
  
  if (draft) {
    // 检查是否过期
    if (Date.now() - draft.timestamp > MAX_DRAFT_AGE) {
      uni.removeStorageSync(`${DRAFT_KEY_PREFIX}${goodsId.value}`);
      console.log(`🗑️ 草稿已过期: ${goodsId.value}`);
      return;
    }
    
    uni.showModal({
      title: "检测到未保存的编辑",
      content: "是否恢复之前的编辑内容？（24小时内有效）",
      success: (res) => {
        if (res.confirm) {
          specForm.value = draft.specForm;
          skuForm.value = draft.skuForm;
          console.log(`✅ 草稿已恢复: ${goodsId.value}`);
        }
        // 使用后清除
        uni.removeStorageSync(`${DRAFT_KEY_PREFIX}${goodsId.value}`);
      }
    });
  }
  
  // 组件卸载时清理
  onUnmounted(() => {
    cleanupExpiredDrafts();
  });
});
```

## 🎯 方案四：完整实用版（推荐）

```
// utils/draftStorage.js
export const DRAFT_KEY_PREFIX = 'goods_stock_draft_';
export const MAX_DRAFT_AGE = 24 * 60 * 60 * 1000; // 24小时

// 获取草稿key
export const getDraftKey = (goodsId) => {
  return `${DRAFT_KEY_PREFIX}${goodsId || 'new'}`;
};

// 保存草稿
export const saveDraft = (goodsId, data) => {
  if (!goodsId) return false;
  
  const draft = {
    goodsId,
    data,
    timestamp: Date.now(),
  };
  
  try {
    uni.setStorageSync(getDraftKey(goodsId), draft);
    console.log(`✅ 草稿保存成功: ${goodsId}`);
    return true;
  } catch (e) {
    console.error('草稿保存失败:', e);
    return false;
  }
};

// 加载草稿
export const loadDraft = (goodsId, autoRemove = true) => {
  if (!goodsId) return null;
  
  const key = getDraftKey(goodsId);
  const draft = uni.getStorageSync(key);
  
  if (!draft) return null;
  
  // 检查是否过期
  if (Date.now() - draft.timestamp > MAX_DRAFT_AGE) {
    uni.removeStorageSync(key);
    console.log(`🗑️ 草稿已过期: ${goodsId}`);
    return null;
  }
  
  // 检查商品ID是否匹配
  if (draft.goodsId !== goodsId) {
    console.warn(`商品ID不匹配: ${draft.goodsId} !== ${goodsId}`);
    return null;
  }
  
  if (autoRemove) {
    uni.removeStorageSync(key);
  }
  
  return draft.data;
};

// 清除草稿
export const clearDraft = (goodsId) => {
  if (!goodsId) return;
  uni.removeStorageSync(getDraftKey(goodsId));
  console.log(`🗑️ 草稿已清除: ${goodsId}`);
};
