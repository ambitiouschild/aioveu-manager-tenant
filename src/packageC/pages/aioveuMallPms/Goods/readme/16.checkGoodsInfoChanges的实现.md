`checkGoodsInfoChanges`的实现和 `checkStockChanges`类似，但更简单。因为 GoodsInfo 的数据结构比 GoodsStock 简单得多。

## 🎯 方案一：简单JSON对比（推荐）

```
// GoodsInfo.vue
// 存储原始数据
let originalGoodsInfo = null;

onMounted(() => {
  // 保存原始数据（深拷贝）
  originalGoodsInfo = JSON.parse(JSON.stringify(goodsInfo.value));
  console.log('📝 保存原始商品信息:', originalGoodsInfo);
});

// 检查是否有修改
const checkGoodsInfoChanges = () => {
  if (!originalGoodsInfo) return false;
  
  const current = goodsInfo.value;
  const original = originalGoodsInfo;
  
  // 使用JSON.stringify简单对比
  const hasChanges = JSON.stringify(current) !== JSON.stringify(original);
  
  if (hasChanges) {
    console.log('🔄 商品信息有修改');
  }
  
  return hasChanges;
};
```

## 🎯 方案二：精确字段对比

```
// GoodsInfo.vue
let originalGoodsInfo = null;

onMounted(() => {
  originalGoodsInfo = { ...goodsInfo.value };
});

// 检查关键字段变化
const checkGoodsInfoChanges = () => {
  if (!originalGoodsInfo) return false;
  
  const current = goodsInfo.value;
  const original = originalGoodsInfo;
  
  // 需要检查的字段列表
  const checkFields = [
    'name',
    'brandId',
    'originPrice',
    'price',
    'picUrl',
    'description',
    'detail'
  ];
  
  // 逐个比较
  for (const field of checkFields) {
    const currentVal = current[field];
    const originalVal = original[field];
    
    // 特殊处理：数字比较
    if (field === 'originPrice' || field === 'price') {
      if (Number(currentVal) !== Number(originalVal)) {
        console.log(`${field} 变化: ${originalVal} → ${currentVal}`);
        return true;
      }
    }
    // 特殊处理：数组比较（轮播图）
    else if (field === 'album') {
      if (!isAlbumEqual(currentVal, originalVal)) {
        console.log('轮播图有变化');
        return true;
      }
    }
    // 普通字段比较
    else if (currentVal !== originalVal) {
      console.log(`${field} 变化: "${originalVal}" → "${currentVal}"`);
      return true;
    }
  }
  
  return false;
};

// 比较轮播图数组
const isAlbumEqual = (arr1, arr2) => {
  if (!Array.isArray(arr1) || !Array.isArray(arr2)) {
    return arr1 === arr2;
  }
  
  if (arr1.length !== arr2.length) return false;
  
  for (let i = 0; i < arr1.length; i++) {
    if (arr1[i] !== arr2[i]) return false;
  }
  
  return true;
};
```

## 🎯 方案四：完整实用版（推荐）

```
// GoodsInfo.vue
import { ref, onMounted, watch } from 'vue';

// 修改状态
const modificationState = ref({
  hasChanges: false,
  modifiedFields: [] as string[],
  lastModified: null as number | null
});

// 原始数据快照
let originalSnapshot: any = null;

// 初始化
onMounted(() => {
  // 保存快照
  originalSnapshot = takeSnapshot(goodsInfo.value);
  console.log('📸 保存原始快照:', originalSnapshot);
  
  // 监听变化
  watch(
    () => goodsInfo.value,
    (newVal) => {
      if (!originalSnapshot) return;
      
      const changes = compareChanges(newVal, originalSnapshot);
      modificationState.value = changes;
      
      if (changes.hasChanges) {
        console.log('🔍 检测到修改:', {
          fields: changes.modifiedFields,
          time: new Date(changes.lastModified!).toLocaleTimeString()
        });
      }
    },
    { deep: true }
  );
});

// 创建数据快照
const takeSnapshot = (data: any) => {
  return JSON.parse(JSON.stringify(data));
};

// 比较变化
const compareChanges = (current: any, original: any) => {
  const modifiedFields: string[] = [];
  
  // 检查的字段配置
  const fieldConfig = [
    { key: 'name', label: '商品名称' },
    { key: 'brandId', label: '品牌' },
    { key: 'originPrice', label: '零售价', isNumber: true },
    { key: 'price', label: '促销价', isNumber: true },
    { key: 'picUrl', label: '主图' },
    { key: 'album', label: '轮播图', isArray: true },
    { key: 'description', label: '商品简介' },
    { key: 'detail', label: '商品详情' }
  ];
  
  fieldConfig.forEach(({ key, label, isNumber, isArray }) => {
    const currentVal = current[key];
    const originalVal = original[key];
    
    let isChanged = false;
    
    if (isNumber) {
      // 数字比较
      isChanged = Number(currentVal) !== Number(originalVal);
    } else if (isArray) {
      // 数组比较
      isChanged = !isArrayEqual(currentVal, originalVal);
    } else {
      // 字符串/其他比较
      isChanged = currentVal !== originalVal;
    }
    
    if (isChanged) {
      modifiedFields.push(label);
    }
  });
  
  return {
    hasChanges: modifiedFields.length > 0,
    modifiedFields,
    lastModified: Date.now()
  };
};

// 比较数组
const isArrayEqual = (arr1: any[] = [], arr2: any[] = []) => {
  if (arr1.length !== arr2.length) return false;
  return arr1.every((item, index) => item === arr2[index]);
};

// 检查是否有修改
const checkGoodsInfoChanges = () => {
  return modificationState.value.hasChanges;
};

// 获取修改详情
const getChangeDetails = () => {
  return {
    hasChanges: modificationState.value.hasChanges,
    fields: modificationState.value.modifiedFields,
    count: modificationState.value.modifiedFields.length
  };
};
```

## 🎯 方案五：优化性能版

```
// GoodsInfo.vue
let originalGoodsInfo = null;
let changeCache = { hasChanges: false, timestamp: 0 };
const CACHE_DURATION = 1000; // 1秒缓存

onMounted(() => {
  originalGoodsInfo = { ...goodsInfo.value };
});

// 带缓存的检查
const checkGoodsInfoChanges = () => {
  const now = Date.now();
  
  // 使用缓存
  if (now - changeCache.timestamp < CACHE_DURATION) {
    return changeCache.hasChanges;
  }
  
  // 重新计算
  if (!originalGoodsInfo) return false;
  
  const hasChanges = JSON.stringify(goodsInfo.value) !== JSON.stringify(originalGoodsInfo);
  
  // 更新缓存
  changeCache = {
    hasChanges,
    timestamp: now
  };
  
  return hasChanges;
};
```

**如果需要更友好的提示**，用方案四，可以告诉用户具体修改了哪些字段。

## 💡 注意点

1. **深拷贝问题**：使用 `JSON.parse(JSON.stringify())`可以处理大部分场景
2. **函数/循环引用**：如果有函数或循环引用，需要特殊处理
3. **性能考虑**：商品信息数据量小，直接 JSON.stringify 没问题
4. **编辑模式**：如果是编辑已有商品，需要保存原始数据
