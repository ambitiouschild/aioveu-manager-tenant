```javascript
  }
);
```

### ж–№жЎҲ1пјҡеҪ»еә•йҮҚеҶҷ GoodsAttribute.vue

```javascript
<script setup lang="ts">
import { ref, computed, watch, onMounted, onUnmounted, nextTick, getCurrentInstance } from "vue";

// иҺ·еҸ–з»„д»¶е®һдҫӢ
const instance = getCurrentInstance();

// Props
const props = defineProps<{
  goodsInfo?: {
    id?: number;
    categoryId?: number;
    attrList?: Array<{ id?: number; name: string; value: string; type?: number }>;
  };
}>();

const emit = defineEmits<{
  (e: "prev"): void;
  (e: "next"): void;
  (e: "update:goodsInfo", value: any): void;
}>();

// ==================== е“Қеә”ејҸж•°жҚ® ====================
// дҪҝз”Ёжң¬ең°зҠ¶жҖҒпјҢдёҚзӣҙжҺҘдҝ®ж”№ props
const localAttrList = ref<Array<{ id?: number; name: string; value: string; type?: number }>>([]);
const errors = ref<Array<{ name?: string; value?: string }>>([]);
const tableHeight = ref<number>(300);
const loading = ref<boolean>(false);
const systemInfo = uni.getSystemInfoSync();

// жҳҜеҗҰжӯЈеңЁеҲқе§ӢеҢ–
const isInitializing = ref<boolean>(true);

// ==================== еҲқе§ӢеҢ– ====================
// д»Һ props еҲқе§ӢеҢ–ж•°жҚ®
const initFromProps = () => {
  console.log("рҹ”Қ еҲқе§ӢеҢ–еұһжҖ§ж•°жҚ®:", props.goodsInfo);
  
  if (props.goodsInfo?.attrList && Array.isArray(props.goodsInfo.attrList)) {
    // дҪҝз”Ё props дёӯзҡ„ж•°жҚ®
    localAttrList.value = props.goodsInfo.attrList.map(attr => ({
      id: attr.id,
      name: attr.name || '',
      value: attr.value || '',
      type: attr.type
    }));
  } else {
    // й»ҳи®ӨдёҖдёӘз©әиЎҢ
    localAttrList.value = [{ name: '', value: '' }];
  }
  
  // еҲқе§ӢеҢ–й”ҷиҜҜж•°з»„
  errors.value = localAttrList.value.map(() => ({}));
  
  console.log("вң… еҲқе§ӢеҢ–е®ҢжҲҗ:", localAttrList.value);
  isInitializing.value = false;
};

// ==================== ж•°жҚ®еҗҢжӯҘ ====================
// еҗҢжӯҘж•°жҚ®еҲ°зҲ¶з»„д»¶
const syncToParent = () => {
  if (isInitializing.value) {
    console.log("вҸёпёҸ еҲқе§ӢеҢ–дёӯпјҢи·іиҝҮеҗҢжӯҘ");
    return;
  }
  
  console.log("рҹ“Ө еҗҢжӯҘж•°жҚ®еҲ°зҲ¶з»„д»¶:", localAttrList.value);
  
  const updatedGoodsInfo = {
    ...props.goodsInfo,
    attrList: localAttrList.value.filter(attr => 
      attr.name.trim() !== '' || attr.value.trim() !== ''
    )
  };
  
  emit("update:goodsInfo", updatedGoodsInfo);
};

// зӣ‘еҗ¬жң¬ең°ж•°жҚ®еҸҳеҢ–пјҢеҗҢжӯҘеҲ°зҲ¶з»„д»¶
watch(
  () => localAttrList.value,
  () => {
    if (!isInitializing.value) {
      syncToParent();
    }
  },
  { deep: true }
);

// зӣ‘еҗ¬зҲ¶з»„д»¶ж•°жҚ®еҸҳеҢ–пјҢеҸӘеңЁйңҖиҰҒж—¶жӣҙж–°
watch(
  () => props.goodsInfo,
  (newValue) => {
    console.log("рҹ“Ҙ зҲ¶з»„д»¶ж•°жҚ®еҸҳеҢ–:", newValue);
    
    // еҰӮжһңжң¬ең°ж•°жҚ®дёәз©әпјҢжүҚд»ҺзҲ¶з»„д»¶еҗҢжӯҘ
    if (localAttrList.value.length === 0 || 
        (localAttrList.value.length === 1 && 
         localAttrList.value[0].name === '' && 
         localAttrList.value[0].value === '')) {
      initFromProps();
    }
  },
  { deep: true, immediate: false } // вқҢ дёҚиҰҒз«ӢеҚіжү§иЎҢ
);

// ==================== дёҡеҠЎж–№жі• ====================
// ж·»еҠ еұһжҖ§
const handleAddAttribute = () => {
  localAttrList.value.push({ name: '', value: '' });
  errors.value.push({});
  
  // ж»ҡеҠЁеҲ°еә•йғЁ
  nextTick(() => {
    calculateTableHeight();
  });
};

// еҲ йҷӨеұһжҖ§
const handleRemoveAttribute = (index: number) => {
  if (localAttrList.value.length <= 1) {
    uni.showToast({
      title: "иҮіе°‘йңҖиҰҒдҝқз•ҷдёҖдёӘеұһжҖ§",
      icon: "none",
      duration: 2000,
    });
    return;
  }
  
  localAttrList.value.splice(index, 1);
  errors.value.splice(index, 1);
};

// еӨ„зҗҶиҫ“е…Ҙ
const handleAttributeInput = (event: any, index: number, field: "name" | "value") => {
  const value = event?.detail?.value || '';
  
  if (index >= 0 && index < localAttrList.value.length) {
    localAttrList.value[index][field] = value;
    
    // жё…йҷӨй”ҷиҜҜ
    if (errors.value[index] && errors.value[index][field]) {
      errors.value[index][field] = '';
    }
  }
};

// йӘҢиҜҒеӯ—ж®ө
const validateField = (field: "name" | "value", index: number) => {
  const value = localAttrList.value[index]?.[field] || '';
  let error = '';
  
  if (!value.trim()) {
    error = `${field === "name" ? "еұһжҖ§еҗҚз§°" : "еұһжҖ§еҖј"}дёҚиғҪдёәз©ә`;
  }
  
  if (errors.value[index]) {
    errors.value[index][field] = error;
  }
};

// йӘҢиҜҒжүҖжңүеӯ—ж®ө
const validateAllFields = () => {
  let isValid = true;
  
  localAttrList.value.forEach((attr, index) => {
    if (!attr.name.trim()) {
      if (!errors.value[index]) errors.value[index] = {};
      errors.value[index].name = "еұһжҖ§еҗҚз§°дёҚиғҪдёәз©ә";
      isValid = false;
    }
    
    if (!attr.value.trim()) {
      if (!errors.value[index]) errors.value[index] = {};
      errors.value[index].value = "еұһжҖ§еҖјдёҚиғҪдёәз©ә";
      isValid = false;
    }
  });
  
  return isValid;
};

// дёӢдёҖжӯҘ
const handleNext = () => {
  if (validateAllFields()) {
    console.log("вң… еұһжҖ§йӘҢиҜҒйҖҡиҝҮ");
    emit("next");
  } else {
    uni.showToast({
      title: "иҜ·еЎ«еҶҷе®Ңж•ҙзҡ„еұһжҖ§дҝЎжҒҜ",
      icon: "none",
      duration: 2000,
    });
  }
};

// дёҠдёҖжӯҘ
const handlePrev = () => {
  emit("prev");
};

// и®Ўз®—иЎЁж јй«ҳеәҰ
const calculateTableHeight = () => {
  if (!instance) return;
  
  nextTick(() => {
    const query = uni.createSelectorQuery().in(instance);
    query
      .select(".component-container")
      .boundingClientRect((data: any) => {
        if (data) {
          const windowHeight = systemInfo.windowHeight;
          const containerTop = data.top || 0;
          const footerHeight = 100;
          const headerHeight = 100;
          tableHeight.value = Math.max(300, windowHeight - containerTop - footerHeight - headerHeight - 20);
        }
      })
      .exec();
  });
};

// ==================== з”ҹе‘Ҫе‘Ёжңҹ ====================
onMounted(() => {
  console.log("рҹ”„ GoodsAttribute з»„д»¶жҢӮиҪҪ");
  
  // еҲқе§ӢеҢ–ж•°жҚ®
  initFromProps();
  
  // и®Ўз®—иЎЁж јй«ҳеәҰ
  calculateTableHeight();
  
  // з»„д»¶жҢӮиҪҪе®ҢжҲҗ
  setTimeout(() => {
    isInitializing.value = false;
  }, 100);
});

onUnmounted(() => {
  console.log("рҹ—‘пёҸ GoodsAttribute з»„д»¶еҚёиҪҪ");
});
</script>
```

```vue
<template>
  <view class="component-container">
    <view class="component-container__main">
      <view class="card-container">

                {{ errors[index].value }}
              </view>
            </view>

            </view>

        </scroll-view>

        <view v-else class="empty-state">
          <text class="empty-text">жҡӮж— е•Ҷе“ҒеұһжҖ§</text>
          <button class="btn-empty-add" @tap="handleAddAttribute">
            ж·»еҠ еұһжҖ§
          </button>
        </view>
      </view>
    </view>

    <view class="action-bar">
      <button class="btn-prev" @tap="handlePrev">
        дёҠдёҖжӯҘпјҢеЎ«еҶҷе•Ҷе“ҒдҝЎжҒҜ
      </button>
      <button class="btn-next" @tap="handleNext">
        дёӢдёҖжӯҘпјҢи®ҫзҪ®е•Ҷе“Ғеә“еӯҳ
      </button>
    </view>
  </view>
</template>
```

3. ж·»еҠ  `isInitializing` ж Үеҝ—пјҢйҳІжӯўеҲқе§ӢеҢ–ж—¶и§ҰеҸ‘жӣҙж–°
4. з®ҖеҢ–ж•°жҚ®жөҒпјҡжң¬ең°зҠ¶жҖҒ вҶ’ зҲ¶з»„д»¶пјҢеҚ•еҗ‘жөҒеҠЁ
