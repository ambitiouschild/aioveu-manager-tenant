`checkStockChanges`需要比较**当前编辑的数据**和**原始数据**。有几种实现方式：

## 🎯 方案一：简单对比（推荐）

```

  return JSON.stringify(obj1) === JSON.stringify(obj2);
```

## 🎯 方案二：详细对比（精确）

```
// 检查规格变化
const checkSpecChanges = () => {
  if (!originalSpecForm.value) return false;
  
  const current = specForm.value;
  const original = originalSpecForm.value;
  
  // 1. 数量变化
  if (current.specList.length !== original.specList.length) {
    console.log('规格数量变化');
    return true;
  }
  
  // 2. 逐个比较
  for (let i = 0; i < current.specList.length; i++) {
    const curSpec = current.specList[i];
    const origSpec = original.specList[i];
    
    // 规格名变化
    if (curSpec.name !== origSpec.name) {
      console.log(`规格${i}名称变化: ${origSpec.name} → ${curSpec.name}`);
      return true;
    }
    
    // 规格值数量变化
    if (curSpec.values.length !== origSpec.values.length) {
      console.log(`规格${i}值数量变化`);
      return true;
    }
    
    // 规格值内容变化
    for (let j = 0; j < curSpec.values.length; j++) {
      const curValue = curSpec.values[j];
      const origValue = origSpec.values[j];
      
      if (curValue.value !== origValue.value || curValue.picUrl !== origValue.picUrl) {
        console.log(`规格${i}值${j}变化`);
        return true;
      }
    }
  }
  
  return false;
};

// 检查SKU变化
const checkSkuChanges = () => {
  if (!originalSkuForm.value) return false;
  
  const current = skuForm.value;
  const original = originalSkuForm.value;
  
  // 1. 数量变化
  if (current.skuList.length !== original.skuList.length) {
    console.log('SKU数量变化');
    return true;
  }
  
  // 2. 逐个比较SKU
  for (let i = 0; i < current.skuList.length; i++) {
    const curSku = current.skuList[i];
    const origSku = original.skuList[i];
    
    // 关键字段对比
    const fields = ['skuSn', 'price', 'stock', 'picUrl', 'specIds'];
    for (const field of fields) {
      if (curSku[field] !== origSku[field]) {
        console.log(`SKU${i} ${field}变化: ${origSku[field]} → ${curSku[field]}`);
        return true;
      }
    }
  }
  
  return false;
};

// 综合检查
const checkStockChanges = () => {
  return checkSpecChanges() || checkSkuChanges();
};
```

## 🎯 方案五：完整实用版（推荐）

```
// GoodsStock.vue
// 状态管理
const modificationState = ref({
  specModified: false,
  skuModified: false,
  lastModified: null
});

// 初始化原始数据
let originalData = null;

onMounted(() => {
  // 保存原始快照
  originalData = {
    specForm: JSON.parse(JSON.stringify(specForm.value)),
    skuForm: JSON.parse(JSON.stringify(skuForm.value))
  };
  
  // 监听规格变化
  watch(
    () => specForm.value,
    (newVal) => {
      if (!originalData) return;
      
      const specChanged = !isEqual(newVal, originalData.specForm);
      if (specChanged !== modificationState.value.specModified) {
        modificationState.value.specModified = specChanged;
        modificationState.value.lastModified = Date.now();
      }
    },
    { deep: true }
  );
  
  // 监听SKU变化
  watch(
    () => skuForm.value,
    (newVal) => {
      if (!originalData) return;
      
      const skuChanged = !isEqual(newVal, originalData.skuForm);
      if (skuChanged !== modificationState.value.skuModified) {
        modificationState.value.skuModified = skuChanged;
        modificationState.value.lastModified = Date.now();
      }
    },
    { deep: true }
  );
});

// 优化的比较函数（忽略临时ID）
const isEqual = (obj1, obj2) => {
  if (obj1 === obj2) return true;
  if (!obj1 || !obj2) return false;
  
  // 对规格特殊处理
  if (obj1.specList && obj2.specList) {
    return isSpecListEqual(obj1.specList, obj2.specList);
  }
  
  // 对SKU特殊处理
  if (obj1.skuList && obj2.skuList) {
    return isSkuListEqual(obj1.skuList, obj2.skuList);
  }
  
  return JSON.stringify(obj1) === JSON.stringify(obj2);
};

// 规格列表比较（忽略临时ID）
const isSpecListEqual = (list1, list2) => {
  if (list1.length !== list2.length) return false;
  
  for (let i = 0; i < list1.length; i++) {
    const s1 = list1[i];
    const s2 = list2[i];
    
    // 比较规格名
    if (s1.name !== s2.name) return false;
    
    // 比较规格值数量
    if (s1.values.length !== s2.values.length) return false;
    
    // 比较每个规格值（比较value和picUrl，忽略id）
    for (let j = 0; j < s1.values.length; j++) {
      const v1 = s1.values[j];
      const v2 = s2.values[j];
      
      if (v1.value !== v2.value || v1.picUrl !== v2.picUrl) {
        return false;
      }
    }
  }
  
  return true;
};

// SKU列表比较
const isSkuListEqual = (list1, list2) => {
  if (list1.length !== list2.length) return false;
  
  for (let i = 0; i < list1.length; i++) {
    const s1 = list1[i];
    const s2 = list2[i];
    
    // 比较关键字段
    const fields = ['skuSn', 'price', 'stock', 'picUrl', 'specIds'];
    for (const field of fields) {
      if (s1[field] !== s2[field]) {
        return false;
      }
    }
  }
  
  return true;
};

// 检查函数
const checkStockChanges = () => {
  return modificationState.value.specModified || modificationState.value.skuModified;
};

// 获取修改详情
const getChangeDetails = () => {
  return {
    hasChanges: checkStockChanges(),
    specModified: modificationState.value.specModified,
    skuModified: modificationState.value.skuModified,
    lastModified: modificationState.value.lastModified
  };
};
```

## ✅ 推荐方案

根据你的需求选择：

1. **快速实现**：方案一（JSON.stringify 对比）
2. **精确控制**：方案二（逐个字段对比）
3. **最佳性能**：方案五（watch + 缓存）
