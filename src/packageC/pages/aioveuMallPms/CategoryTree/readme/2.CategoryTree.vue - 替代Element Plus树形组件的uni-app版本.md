<script setup lang="ts">
import { ref, reactive, onMounted, onUnmounted, defineEmits, defineExpose, nextTick } from 'vue'
import CategoryTreeNode from './CategoryTreeNode.vue'
import type { Ref } from 'vue'

// 定义类型
interface CategoryNode {
  id: number
  name: string
  level: number
  parentId?: number
  iconUrl?: string
  visible?: number
  sort?: number
  children?: CategoryNode[]
  [key: string]: any
}

interface FormData {
  id?: number
  name: string
  parentId: number
  level?: number
  iconUrl: string
  visible: number
  sort: number
}

interface ParentCategory {
  id: number
  name: string
  level: number
}

// 定义组件事件
const emit = defineEmits<{
  (e: 'category-click', data: CategoryNode | null): void
}>()

// 响应式数据
const showCategoryTree = ref<boolean>(true)
const loading = ref<boolean>(true)
const categoryOptions = ref<CategoryNode[]>([])
const listHeight = ref<string>('600rpx')
const expandedNodes = ref<Set<number>>(new Set())

// 对话框控制
const popupRef = ref<any>(null)
const deletePopupRef = ref<any>(null)
const dialogTitle = ref<string>('')
const popupType = ref<string>('info')

// 表单数据
const formData = reactive<FormData>({
  id: undefined,
  name: '',
  parentId: 0,
  level: undefined,
  iconUrl: '',
  visible: 1,
  sort: 1
})

// 错误信息
const errors = reactive<Record<string, string>>({
  name: ''
})

// 父级分类
const parentCategory = reactive<ParentCategory>({
  id: 0,
  name: '全部分类',
  level: 0
})

// 当前操作分类
const currentCategory = ref<CategoryNode | null>(null)
const categoryToDelete = ref<CategoryNode | null>(null)

// 显示状态选项
const visibleOptions = [
  { label: '显示', value: 1 },
  { label: '隐藏', value: 0 }
]

/**
 * 计算列表高度
 */
const calculateListHeight = () => {
  uni.getSystemInfo({
    success: (res) => {
      const windowHeight = res.windowHeight
      const statusBarHeight = res.statusBarHeight || 0
      const bottomSafeArea = 50
      listHeight.value = `${windowHeight - statusBarHeight - bottomSafeArea}px`
    }
  })
}

/**
 * 查询分类数据
 */
const handleQuery = async () => {
  loading.value = true
  try {
    // 这里替换为你的实际API调用
    const response = await new Promise<CategoryNode[]>((resolve) => {
      setTimeout(() => {
        resolve([
          {
            id: 1,
            name: '电子产品',
            parentId: 0,
            level: 1,
            children: [
              {
                id: 2,
                name: '手机',
                parentId: 1,
                level: 2,
                children: [
                  {
                    id: 3,
                    name: '智能手机',
                    parentId: 2,
                    level: 3,
                    iconUrl: '',
                    children: []
                  }
                ]
              }
            ]
          }
        ])
      }, 500)
    })
    
    // 构建树形数据，添加虚拟根节点"全部分类"
    categoryOptions.value = [{
      id: 0,
      name: '全部分类',
      parentId: 0,
      level: 0,
      children: response
    }]
    
    // 默认展开一级分类
    expandedNodes.value.add(0)
  } catch (error) {
    console.error('查询分类数据失败:', error)
    uni.showToast({
      title: '获取分类数据失败',
      icon: 'none'
    })
  } finally {
    loading.value = false
  }
}

/**
 * 节点点击事件处理
 */
const onNodeClick = (node: CategoryNode) => {
  // 保存当前分类信息
  currentCategory.value = { ...node }
  
  // 如果是根节点，设置父级为全部分类
  if (node.id === 0) {
    parentCategory.id = 0
    parentCategory.name = '全部分类'
    parentCategory.level = 0
  } else {
    // 查找父节点
    const findParent = (categories: CategoryNode[], targetId: number): CategoryNode | null => {
      for (const category of categories) {
        if (category.id === targetId) {
          return category
        }
        if (category.children && category.children.length > 0) {
          const found = findParent(category.children, targetId)
          if (found) return found
        }
      }
      return null
    }
    
    const parentNode = findParent(categoryOptions.value, node.parentId || 0)
    if (parentNode) {
      parentCategory.id = parentNode.id
      parentCategory.name = parentNode.name
      parentCategory.level = parentNode.level
    } else {
      parentCategory.id = 0
      parentCategory.name = '全部分类'
      parentCategory.level = 0
    }
  }
  
  // 向父组件发射事件
  emit('category-click', node)
}

/**
 * 切换节点展开状态
 */
const onToggleExpand = (nodeId: number) => {
  if (expandedNodes.value.has(nodeId)) {
    expandedNodes.value.delete(nodeId)
  } else {
    expandedNodes.value.add(nodeId)
  }
}

/**
 * 新增分类
 */
const onAdd = (node: CategoryNode) => {
  dialogTitle.value = '新增商品分类'
  popupType.value = 'info'
  
  // 重置表单
  resetForm()
  formData.id = undefined
  
  if (node && node.id != null) {
    // 设置父级分类
    parentCategory.id = node.id
    parentCategory.name = node.name
    parentCategory.level = node.level
  }
  
  // 显示弹窗
  popupRef.value?.open()
}

/**
 * 修改分类
 */
const onUpdate = (node: CategoryNode) => {
  // 先触发节点点击，获取父节点信息
  onNodeClick(node)
  
  dialogTitle.value = '修改商品分类'
  popupType.value = 'info'
  
  // 将当前分类数据复制到表单中
  Object.assign(formData, {
    id: node.id,
    name: node.name,
    parentId: node.parentId || 0,
    level: node.level,
    iconUrl: node.iconUrl || '',
    visible: node.visible || 1,
    sort: node.sort || 1
  })
  
  // 显示弹窗
  popupRef.value?.open()
}

/**
 * 删除分类
 */
const onDelete = (node: CategoryNode) => {
  categoryToDelete.value = node
  deletePopupRef.value?.open()
}

/**
 * 确认删除
 */
const confirmDelete = async () => {
  try {
    if (!categoryToDelete.value) return
    
    // 模拟API调用
    await new Promise((resolve) => {
      setTimeout(resolve, 500)
    })
    
    uni.showToast({
      title: '删除成功',
      icon: 'success'
    })
    
    handleQuery()
  } catch (error) {
    console.error('删除失败:', error)
    uni.showToast({
      title: '删除失败',
      icon: 'none'
    })
  } finally {
    categoryToDelete.value = null
  }
}

/**
 * 取消删除
 */
const cancelDelete = () => {
  categoryToDelete.value = null
}

/**
 * 选择图片
 */
const chooseImage = () => {
  uni.chooseImage({
    count: 1,
    sizeType: ['compressed'],
    sourceType: ['album', 'camera'],
    success: (res) => {
      formData.iconUrl = res.tempFilePaths[0]
    }
  })
}

/**
 * 提交表单
 */
const submitForm = async () => {
  // 验证表单
  if (!formData.name) {
    errors.name = '请输入分类名称'
    return
  } else {
    errors.name = ''
  }
  
  try {
    if (formData.id) {
      // 修改操作
      await new Promise((resolve) => {
        setTimeout(resolve, 500)
      })
      
      uni.showToast({
        title: '修改成功',
        icon: 'success'
      })
    } else {
      // 新增操作
      const newData = {
        ...formData,
        parentId: parentCategory.id,
        level: (parentCategory.level || 0) + 1
      }
      
      await new Promise((resolve) => {
        setTimeout(resolve, 500)
      })
      
      uni.showToast({
        title: '新增成功',
        icon: 'success'
      })
    }
    
    closeDialog()
    handleQuery()
  } catch (error) {
    console.error('提交表单失败:', error)
    uni.showToast({
      title: '提交失败',
      icon: 'none'
    })
  }
}

/**
 * 重置表单
 */
const resetForm = () => {
  Object.assign(formData, {
    id: undefined,
    name: '',
    parentId: 0,
    level: undefined,
    iconUrl: '',
    visible: 1,
    sort: 1
  })
  
  Object.assign(parentCategory, {
    id: 0,
    name: '全部分类',
    level: 0
  })
  
  errors.name = ''
}

/**
 * 关闭对话框
 */
const closeDialog = () => {
  popupRef.value?.close()
  resetForm()
}

/**
 * 清理方法
 */
const cleanup = () => {
  console.log('🧹 CategoryTree 清理')
  showCategoryTree.value = false
  expandedNodes.value.clear()
  resetForm()
}

// 组件挂载时加载数据
onMounted(() => {
  calculateListHeight()
  handleQuery()
})

// 监听窗口大小变化
onMounted(() => {
  uni.onWindowResize(() => {
    calculateListHeight()
  })
})

// 组件卸载时
onUnmounted(() => {
  cleanup()
})

// 暴露方法给父组件
defineExpose({
  cleanup,
  handleQuery
})
</script>

<template>
  <view>
    <view 
      class="tree-node" 
      :class="['level-' + level, { 'has-children': node.children && node.children.length > 0 }]"
      :style="{ paddingLeft: (level * 20) + 'px' }"
    >
      <view class="node-content" @click="onNodeClick(node)">
        <!-- 展开/收起图标 -->

          <text v-if="expanded" class="icon">▼</text>
          <text v-else class="icon">▶</text>
        </view>
        <view v-else class="expand-placeholder"></view>

          v-if="node.level === 3 && node.iconUrl" 
          :src="node.iconUrl" 
          class="node-icon" 
          mode="aspectFit"
        />
        <view v-else-if="node.level === 3" class="icon-placeholder">
          <text class="icon">📁</text>
        </view>
        
        <!-- 分类名称 -->
        <text class="node-name">{{ node.name }}</text>
        
        <!-- 操作按钮 -->
        <view class="action-buttons" @click.stop>
          <!-- 非三级分类显示新增按钮 -->
          <view 
            v-if="node.level !== 3" 
            class="action-button add-button"
            @click="$emit('add', node)"
          >
            新增
          </view>
          
          <!-- 非根节点显示编辑按钮 -->
          <view 
            v-if="node.id !== 0" 
            class="action-button edit-button"
            @click="$emit('update', node)"
          >
            编辑
          </view>
          
          <!-- 叶子节点显示删除按钮 -->
          <view 
            v-if="node.id && (!node.children || node.children.length === 0)" 
            class="action-button delete-button"
            @click="$emit('delete', node)"
          >
            删除
          </view>
        </view>
      </view>
    </view>
    
    <!-- 子节点 -->
    <view v-if="expanded && node.children && node.children.length > 0" class="children-container">
      <category-tree-node
        v-for="child in node.children"
        :key="child.id"
        :node="child"
        :level="level + 1"
        :expanded-nodes="expandedNodes"
        @node-click="$emit('node-click', $event)"
        @toggle-expand="$emit('toggle-expand', $event)"
        @add="$emit('add', $event)"
        @update="$emit('update', $event)"
        @delete="$emit('delete', $event)"
      />
    </view>
  </view>
</template>

<script setup lang="ts">
import { ref, computed, defineProps, defineEmits } from 'vue'

// 定义类型
interface CategoryNode {
  id: number
  name: string
  level: number
  parentId?: number
  iconUrl?: string
  children?: CategoryNode[]
  [key: string]: any
}

// 定义props
const props = defineProps<{
  node: CategoryNode
  level: number
  expandedNodes: Set<number>
}>()

// 定义事件
const emit = defineEmits<{
  (e: 'node-click', data: CategoryNode): void
  (e: 'toggle-expand', nodeId: number): void
  (e: 'add', data: CategoryNode): void
  (e: 'update', data: CategoryNode): void
  (e: 'delete', data: CategoryNode): void
}>()

// 计算展开状态
const expanded = computed(() => {
  return props.expandedNodes.has(props.node.id)
})

// 切换展开状态
const toggleExpand = () => {
  emit('toggle-expand', props.node.id)
}

// 节点点击
const onNodeClick = (node: CategoryNode) => {
  emit('node-click', node)
}
</script>
