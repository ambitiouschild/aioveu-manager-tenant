import { ref, computed, watch } from 'vue'

// 定义基础节点类型
interface BaseTreeNode {
  id: number
  name: string
  level: number
  parentId?: number
  iconUrl?: string
  visible?: number
  sort?: number
  children?: BaseTreeNode[]
  [key: string]: any
}

// 定义扁平化节点类型
interface FlatTreeNode {
  id: number
  name: string
  level: number
  parentId?: number
  iconUrl?: string
  visible?: number
  sort?: number
  children?: BaseTreeNode[]
  expanded: boolean
  _visible: boolean
  hasChildren: boolean
  _parentIds: number[]
  [key: string]: any
}

const props = defineProps<{
  data: BaseTreeNode[]
}>()

const emit = defineEmits<{
  nodeClick: [node: BaseTreeNode]
  add: [node: BaseTreeNode]
  update: [node: BaseTreeNode]
  delete: [node: BaseTreeNode]
}>()

// 存储扁平化的节点
const flatNodes = ref<FlatTreeNode[]>([])
// 选中的节点ID
const selectedId = ref<number>()

/**
 * 递归扁平化树结构
 */
const flattenTree = (nodes: BaseTreeNode[], parentIds: number[] = [], level: number = 0): FlatTreeNode[] => {
  const result: FlatTreeNode[] = []
  
  for (const node of nodes) {
    const currentParentIds = [...parentIds, node.id]
    const hasChildren = !!(node.children && node.children.length > 0)
    
    const flatNode: FlatTreeNode = {
      ...node,
      expanded: false,  // 默认不展开
      _visible: true,   // 默认可见，用 _visible 避免和 visible 属性冲突
      hasChildren,
      _parentIds: currentParentIds,  // 记录所有父级ID
      level
    }
    
    result.push(flatNode)
    
    // 递归处理子节点
    if (hasChildren) {
      result.push(...flattenTree(node.children!, currentParentIds, level + 1))
    }
  }
  
  return result
}

/**
 * 监听数据变化，初始化扁平化节点
 */
watch(() => props.data, (newData) => {
  if (newData && newData.length > 0) {
    flatNodes.value = flattenTree(newData)
    console.log('📊 扁平化后的节点:', flatNodes.value)
  }
}, { immediate: true, deep: true })

/**
 * 计算显示的节点列表
 */
const displayList = computed(() => {
  const result: FlatTreeNode[] = []
  
  for (const node of flatNodes.value) {
    // 检查节点是否应该显示
    let shouldShow = true
    
    // 检查所有父节点是否展开
    for (let i = 0; i < node._parentIds.length - 1; i++) {
      const parentId = node._parentIds[i]
      const parentNode = flatNodes.value.find(n => n.id === parentId)
      
      // 如果某个父节点未展开，则该节点不显示
      if (parentNode && !parentNode.expanded) {
        shouldShow = false
        break
      }
    }
    
    if (shouldShow) {
      result.push(node)
    }
  }
  
  return result
})

/**
 * 展开/收起节点
 */
const toggleExpand = (node: FlatTreeNode) => {
  if (node.hasChildren) {
    node.expanded = !node.expanded
    console.log(`切换节点 ${node.name} 展开状态: ${node.expanded}`)
  }
}

/**
 * 点击节点
 */
const handleClick = (node: BaseTreeNode) => {
  selectedId.value = node.id
  emit('nodeClick', node)
}

/**
 * 查询分类数据
 */
const handleQuery = async () => {
  loading.value = true
  try {
    const response = await PmsCategoryAPI.getListCategories({})
    
    console.log('📊 原始数据:', response)
    
    if (Array.isArray(response)) {
      // 存储原始数据
      rawCategoryList.value = response
      
      // 构建树
      const tree = buildTree(rawCategoryList.value)
      
      // 统计各层级节点数 - 修复类型
      interface LevelCount {
        [key: number]: number
      }
      const countByLevel: LevelCount = { 0: 0, 1: 0, 2: 0, 3: 0 }
      
      const countNodes = (nodes: CategoryNode[]) => {
        nodes.forEach(node => {
          // 使用类型断言确保安全
          const levelKey = node.level as keyof LevelCount
          if (countByLevel[levelKey] !== undefined) {
            countByLevel[levelKey]++
          } else {
            countByLevel[node.level] = 1
          }
          
          if (node.children) {
            countNodes(node.children)
          }
        })
      }
      
      countNodes(tree)
      console.log('📈 各层级节点数:', countByLevel)
      console.log('🌳 构建的树结构:', tree)
      
    } else {
      console.error('返回数据格式不正确:', response)
    }
  } catch (error) {
    console.error('查询分类数据失败:', error)
  } finally {
    loading.value = false
  }
}

## 🎯 关键修改点：

1. **分离类型定义**：
   - `BaseTreeNode`：原始数据接口
   - `FlatTreeNode`：扁平化后的节点接口，包含扩展属性
   - 使用 `_visible`替代 `visible`，避免和原始数据的 `visible`（number 类型）冲突
2. **修复索引签名错误**：
   - 使用 `interface LevelCount { [key: number]: number }`定义可扩展的数字索引
   - 使用类型断言 `as keyof LevelCount`确保类型安全
3. **简化代码**：
   - 移除了不必要的操作按钮（保持简洁）
   - 确保所有类型定义正确
