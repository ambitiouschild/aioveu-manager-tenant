您说得对！在 UniApp 中，**不能使用 `uni.request`来上传文件**，必须使用 `uni.uploadFile`。这是一个常见的误区。

## 📌 UniApp 上传文件的正确方式

### ❌ 错误的方式（不能工作）

```
// 错误！uni.request 不能上传文件
uni.request({
  url: '/api/upload',
  method: 'POST',
  data: formData,  // 不能传文件
  header: {
    'Content-Type': 'multipart/form-data'
  }
});
```

### ✅ 正确的方式

```
// 正确！必须使用 uni.uploadFile
uni.uploadFile({
  url: '/api/upload',  // 上传接口地址
  filePath: tempFilePath,  // 临时文件路径
  name: 'file',  // 后端接收的文件字段名
  formData: {  // 额外的表单数据
    category: 'product',
    type: 'main'
  },
  header: {  // 请求头
    'Authorization': 'Bearer ' + uni.getStorageSync('token')
  },
  success: (res) => {
    console.log('上传成功:', res);
  },
  fail: (err) => {
    console.error('上传失败:', err);
  }
});
```

## 🔧 修复您的 FileAPI.ts

```
// file/file.ts
import type { FileInfo, ApiResponse } from "@/types/api";

// 配置文件上传
const FILE_UPLOAD_URL = '/aioveu-tenant/api/v1/files';

interface UploadOptions {
  filePath: string;
  category?: string;
  type?: string;
  businessId?: string | number;
  fileName?: string;
  onProgress?: (percent: number) => void;
}

const FileAPI = {
  /**
   * UniApp 文件上传
   * 必须使用 uni.uploadFile
   */
  uploadUniFile(options: UploadOptions): Promise<ApiResponse<FileInfo>> {
    return new Promise((resolve, reject) => {
      const { filePath, onProgress, ...formData } = options;
      
      // 验证文件路径
      if (!filePath || typeof filePath !== 'string') {
        reject(new Error('无效的文件路径'));
        return;
      }
      
      console.log('📤 开始上传文件:', {
        文件路径: filePath,
        上传地址: FILE_UPLOAD_URL,
        表单数据: formData
      });
      
      // 获取认证 token
      const token = uni.getStorageSync('token') || '';
      console.log('🔑 认证token:', token ? '已设置' : '未设置');
      
      // 使用 uni.uploadFile
      const uploadTask = uni.uploadFile({
        url: FILE_UPLOAD_URL,
        filePath: filePath,
        name: 'file',  // 后端接收文件的字段名
        formData: {
          category: formData.category || 'product',
          type: formData.type || 'main',
          ...(formData.businessId && { businessId: formData.businessId }),
          ...(formData.fileName && { fileName: formData.fileName })
        },
        header: {
          'Authorization': token ? `Bearer ${token}` : '',
          'Content-Type': 'multipart/form-data'
        },
        success: (uploadRes) => {
          console.log('📥 上传响应:', {
            状态码: uploadRes.statusCode,
            响应数据: uploadRes.data
          });
          
          if (uploadRes.statusCode === 200) {
            try {
              const data = JSON.parse(uploadRes.data) as ApiResponse<FileInfo>;
              
              if (data.code === 0 && data.data) {
                console.log('✅ 上传成功:', data.data.url);
                resolve(data);
              } else {
                console.error('❌ 业务逻辑失败:', data.msg);
                reject(new Error(data.msg || '上传失败'));
              }
            } catch (parseError) {
              console.error('❌ 解析响应失败:', parseError, uploadRes.data);
              reject(new Error('解析响应失败'));
            }
          } else {
            console.error(`❌ HTTP错误，状态码: ${uploadRes.statusCode}`);
            reject(new Error(`上传失败，状态码: ${uploadRes.statusCode}`));
          }
        },
        fail: (error) => {
          console.error('❌ 上传请求失败:', error);
          reject(error);
        }
      });
      
      // 监听上传进度
      if (onProgress) {
        uploadTask.onProgressUpdate((res) => {
          console.log(`📈 上传进度: ${res.progress}%`);
          onProgress(res.progress);
        });
      }
    });
  },

  /**
   * 批量上传文件
   */
  async uploadMultiple(files: string[], options: Omit<UploadOptions, 'filePath'> = {}) {
    console.log('🖼️ 开始批量上传，数量:', files.length);
    
    const results = await Promise.allSettled(
      files.map(filePath => 
        this.uploadUniFile({ filePath, ...options })
          .then(res => ({ success: true, data: res.data, filePath }))
          .catch(error => ({ success: false, error, filePath }))
      )
    );
    
    const successful: FileInfo[] = [];
    const failed: { file: string; error: any }[] = [];
    
    results.forEach((result, index) => {
      if (result.status === 'fulfilled' && result.value.success) {
        successful.push(result.value.data);
      } else {
        const filePath = files[index];
        const error = result.status === 'fulfilled' 
          ? result.value.error 
          : result.reason;
        
        failed.push({ file: filePath, error });
      }
    });
    
    return { successful, failed };
  },

  /**
   * Web 环境上传（使用 XMLHttpRequest）
   * 保留给 H5 使用
   */
  uploadWebFile(formData: FormData, onProgress?: (percent: number) => void) {
    return new Promise<ApiResponse<FileInfo>>((resolve, reject) => {
      const xhr = new XMLHttpRequest();
      
      xhr.open('POST', FILE_UPLOAD_URL);
      
      const token = uni.getStorageSync('token');
      if (token) {
        xhr.setRequestHeader('Authorization', `Bearer ${token}`);
      }
      
      if (onProgress) {
        xhr.upload.onprogress = (event) => {
          if (event.lengthComputable) {
            const percent = Math.round((event.loaded * 100) / event.total);
            onProgress(percent);
          }
        };
      }
      
      xhr.onload = () => {
        if (xhr.status === 200) {
          try {
            const data = JSON.parse(xhr.responseText) as ApiResponse<FileInfo>;
            resolve(data);
          } catch (error) {
            reject(new Error('解析响应失败'));
          }
        } else {
          reject(new Error(`上传失败，状态码: ${xhr.status}`));
        }
      };
      
      xhr.onerror = () => {
        reject(new Error('网络错误'));
      };
      
      xhr.send(formData);
    });
  },

  // 其他方法...
};

export default FileAPI;
```

## 🎯 关键点总结：

1. **必须使用 `uni.uploadFile`** 而不是 `uni.request`
2. **filePath 参数** 是必须的
3. **name 参数** 指定后端接收的字段名
4. **formData 参数** 传递额外数据
5. **header 中不需要设置 Content-Type**，UniApp 会自动处理
6. **可以监听上传进度** 通过 `onProgressUpdate`

这样修改后，您的文件上传应该可以正常工作了！
